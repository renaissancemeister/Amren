	placeSbarTop = function(){
		var t=$(window).scrollTop()+($('#topbar').height()-60);
		var articleOffset=$("#main-story").offset().top - t;
		var articleEnd=$("#main-story").height()-400;
		var newArticleEnd=articleEnd+$("#main-story").offset().top;
		var t=$(window).scrollTop()+($('#topbar').height()+90);
		if (articleOffset>100)    { $('#shareBarWrapper').css({'top':articleOffset,'position':'fixed','margin-top':0}); }
		else if (t>newArticleEnd) { $('#shareBarWrapper').css({'position':'absolute','margin-top':newArticleEnd}); }
		else                      { $('#shareBarWrapper').css({'top':100,'position':'fixed','margin-top':0}); }
	}
	placeSbar = function(){
		var divPos     = $("#main-story").offset();
		var sBarWidth  = $('#shareBarWrapper').outerWidth();
		var newLeftPos = divPos.left - sBarWidth -70;
		$('#shareBarWrapper').css('left', newLeftPos);
		$('#shareBarWrapper').css('display','block');
		$('#shareBarWrapper').removeClass('shareBarHide');
	}
	//placeSbarTop must run before placeSbar
